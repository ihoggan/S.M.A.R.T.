        pattern = {
            'total_attacks': len(attacks),

            'interval_std': intervals.std(),

            'targets': attacks['destination'].nunique()
        }
        
        attack_patterns[attacker] = pattern
    
    return attack_patterns

üìç Signal Strength Analysis

# Estimating device distance from signal strength
def estimate_distance(signal_dbm, frequency_mhz=2400):
    # Free Space Path Loss model
    tx_power = 20  # Typical router TX power (dBm)
    path_loss = tx_power - signal_dbm
    
    # Convert to distance (meters)
    frequency_hz = frequency_mhz * 1e6
    distance = 10 ** ((path_loss - 27.55) / 20) / frequency_hz * 299792458
    
    return distance
