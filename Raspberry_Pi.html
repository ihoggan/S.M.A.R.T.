Create Your First Hardware Python Script

Create a file called blink_led.py:

import RPi.GPIO as GPIO import time # Set up GPIO GPIO.setmode(GPIO.BCM) GPIO.setup(18, GPIO.OUT) try: # Blink LED 10 times for i in range(10): GPIO.output(18, GPIO.HIGH) # Turn on print(f"LED ON - Blink {i+1}") time.sleep(0.5) GPIO.output(18, GPIO.LOW) # Turn off print(f"LED OFF - Blink {i+1}") time.sleep(0.5) finally: GPIO.cleanup() # Always clean up GPIO settings print("GPIO cleanup complete!")
Run Your Script
python3 blink_led.py

Advanced: Button-Controlled LED

Add a button to control your LED interactively:

üîò Button Wiring (add to existing LED circuit): Raspberry Pi Pin 7 (GPIO4) ‚îÄ‚îÄ Button ‚îÄ‚îÄ Raspberry Pi 3.3V (Pin 1) Raspberry Pi Pin 7 (GPIO4) ‚îÄ‚îÄ 10kŒ© Resistor ‚îÄ‚îÄ Ground
import RPi.GPIO as GPIO import time # Pin setup LED_PIN = 18 BUTTON_PIN = 4 GPIO.setmode(GPIO.BCM) GPIO.setup(LED_PIN, GPIO.OUT) GPIO.setup(BUTTON_PIN, GPIO.IN, pull_up_down=GPIO.PUD_DOWN) print("Press button to toggle LED (Ctrl+C to exit)") led_state = False try: while True: if GPIO.input(BUTTON_PIN) == GPIO.HIGH: # Button pressed - toggle LED led_state = not led_state GPIO.output(LED_PIN, led_state) if led_state: print("üí° LED ON!") else: print("üî¥ LED OFF!") # Wait for button release while GPIO.input(BUTTON_PIN) == GPIO.HIGH: time.sleep(0.01) time.sleep(0.1) # Debounce delay time.sleep(0.01) # Small delay to prevent excessive CPU usage except KeyboardInterrupt: print("\nExiting...") finally: GPIO.cleanup()

Build a web-based temperature monitoring system:

Step 3: Create Temperature Web Server

Create temp_server.py:

from flask import Flask, render_template_string from w1thermsensor import W1ThermSensor import time import threading from datetime import datetime app = Flask(__name__) current_temp = 0 last_update = "" def read_temperature(): global current_temp, last_update try: sensor = W1ThermSensor() while True: current_temp = round(sensor.get_temperature(), 1) last_update = datetime.now().strftime("%H:%M:%S") time.sleep(10) # Read every 10 seconds except: current_temp = "Sensor not found" last_update = "Error" # Start temperature reading in background temp_thread = threading.Thread(target=read_temperature) temp_thread.daemon = True temp_thread.start() @app.route('/') def index(): html = ''' üå°Ô∏è Raspberry Pi Temperature Monitor

üå°Ô∏è Temperature Monitor

{{ temp }}¬∞C

Last Update: {{ update_time }}

Updates every 10 seconds

''' return render_template_string(html, temp=current_temp, update_time=last_update) if __name__ == '__main__': print("üåê Starting temperature web server...") print("üì± Visit http://your-pi-ip:5000 in your browser") app.run(host='0.0.0.0', port=5000)

Step 4: Run Your Web Server
